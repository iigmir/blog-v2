如何避免程式短時間內請求過多 HTTP

可以用 throttle 與 debounce：

/**
 * @see [throttle](https://www.explainthis.io/zh-hant/swe/throttle)
 */
function throttle(fn, delay = 500) {
    let timer = null;
    return (...args) => {
        if (timer) {
            return;
        }
        timer = setTimeout(() => {
            fn(...args);
            timer = null;
        }, delay);
    };
}

/**
 * @see [debounce](https://www.explainthis.io/zh-hant/swe/debounce)
 */
function debounce(fn, delay = 500) {
    let timer = null;
    return (...args) => {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn(...args);
        }, delay);
    };
}
